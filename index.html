from transformers import AutoModelForCausalLM, AutoTokenizer
import tkinter as tk

# Load AI model and tokenizer
model_name = 'gpt2'
tokenizer = AutoTokenizer.from_pretrained(model_name)
model = AutoModelForCausalLM.from_pretrained(model_name)

# Function to chat with Xenon
def chat_with_xenon():
    user_input = entry.get()
    if user_input.lower() == "exit":
        root.destroy()
    else:
        inputs = tokenizer.encode(user_input, return_tensors='pt')
        outputs = model.generate(inputs, max_length=100, num_return_sequences=1)
        response = tokenizer.decode(outputs[0], skip_special_tokens=True)
        chatbox.insert(tk.END, f"You: {user_input}\n")
        chatbox.insert(tk.END, f"Xenon: {response}\n\n")
        entry.delete(0, tk.END)

# GUI Setup
root = tk.Tk()
root.title("Xenon Chatbox")

chatbox = tk.Text(root, height=20, width=50)
chatbox.pack(pady=10)

entry = tk.Entry(root, width=50)
entry.pack(pady=5)

send_button = tk.Button(root, text="Send", command=chat_with_xenon)
send_button.pack(pady=5)

root.mainloop()
